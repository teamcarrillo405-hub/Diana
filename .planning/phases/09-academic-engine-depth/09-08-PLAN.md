<objective>
Ship F13 (parent share) + F14 (teacher snapshot) as student-initiated read-only public views. Student generates a share link in Settings. Link is valid for 7 days. Public view requires no auth. Parent view shows completion + upcoming + study minutes; teacher view shows per-class AI mode + reading font + accommodation note + time multiplier. The public route lives OUTSIDE `app/(app)/` and uses the service-role client. ALL downstream queries are scoped to `owner_id` from the token row — never query params.

Purpose: A student with ADHD/dyslexia builds trust with parents and teachers by sharing _what's happening this week_ (effort, accommodations) without exposing diagnosis codes, AI tool names, or assignment content. The link is theirs to revoke at any time.

Output: New `share_links` table (migration 0016 — NOT 0015, which is reserved for LMS in 09-05), `lib/sharing/` module, public `/share/[token]` route with parent + teacher variants, and a Sharing section added to Settings.

Wave 4 depends on Wave 3 completion (LMS + calendar) but has NO direct file overlap with 09-07 (AI transparency wave-4 sibling), so 09-08 and 09-07 run in parallel.

Design decisions LOCKED:
- D-01 — Migration number is `0016_share_links.sql` (NOT `0015`; 0015 is reserved by 09-05 LMS plan; 0014 is `tts_provider.sql`).
- D-02 — Token column is `text UNIQUE NOT NULL DEFAULT gen_random_uuid()::text` (URL-safe; gen_random_uuid is available in Supabase pgcrypto).
- D-03 — RLS policy allows owner to manage their own rows; NO public read policy. The public route uses service-role client to bypass RLS for token lookup.
- D-04 — Public route lives at `app/share/[token]/page.tsx` (OUTSIDE `app/(app)/` so the (app) layout's auth+onboarding redirect does NOT fire).
- D-05 — Service-role client factory is `lib/supabase/service.ts` (NEW; never used by browser code; server-only).
- D-06 — Token lookup query: `.from('share_links').select('*').eq('token', token).is('revoked_at', null).gt('expires_at', new Date().toISOString()).maybeSingle()`. All three conditions enforced server-side. Failure on any → calm fallback page.
- D-07 — Owner ID is read from the share_links row ONLY. NEVER from query params, never from a header. Pitfall: trusting query params is the classic shared-link auth bypass.
- D-08 — Parent summary "this week" = Monday 00:00 local→now in profile timezone. Implementation uses ISO week start computed from `new Date()` and `profile.timezone` (fallback UTC). Keep simple: compute startOfWeek server-side in the parent-summary component.
- D-09 — Parent summary completed count comes from `task_signals WHERE kind='completed' AND owner_id=<token.owner_id> AND created_at >= startOfWeek`. Use task_signals (not assignments.status) because signals are the canonical event log per Phase 2 (STATE.md 02-02 decision).
- D-10 — Parent summary study time comes from `time_logs WHERE owner_id=<token.owner_id> AND opened_at >= startOfWeek AND closed_at IS NOT NULL`, summed as `(closed_at - opened_at)` milliseconds.
- D-11 — Teacher snapshot per-class data: `classes.name`, `classes.ai_mode` (Phase 6 added column). Profile-level: `profiles.reading_font`, `profiles.disability_flags` (dyslexia bit), `profiles.extra_time_pct`. The dyslexia flag is rendered as the literal phrase "May benefit from extended reading time" (NEVER "has dyslexia").
- D-12 — Calm copy invariants: "X assignments finished this week" (NOT "X out of Y"); "May benefit from extended reading time" (NOT diagnosis); "This link works until [date]" (NOT "expires in X days"); no streak language; no exclamations.
- D-13 — Public pages render minimal layout (no header nav, no app chrome). They are intentionally austere to feel like a printout, not a social profile.
- D-14 — No test for the public route (requires DB integration). Tests cover createShareLink + revokeShareLink only.
- D-15 — Service-role client uses `SUPABASE_SERVICE_ROLE_KEY` env var; if missing, the public route returns the calm fallback (treat missing env as "no longer active" for safety).

    ```typescript
    "use server";

    import { createClient } from "@/lib/supabase/server";
    import { revalidatePath } from "next/cache";
    import type { ShareLink, ShareType } from "./types";

    export async function createShareLink(
      shareType: ShareType,
    ): Promise<{ token: string; expiresAt: string } | { error: string }> {
      if (shareType !== "parent_summary" && shareType !== "teacher_snapshot") {
        return { error: "Invalid share type." };
      }
      const supabase = await createClient();
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) return { error: "Not signed in." };

      const { data, error } = await supabase
        .from("share_links")
        .insert({ owner_id: user.id, share_type: shareType })
        .select("token, expires_at")
        .single();

      if (error || !data) return { error: "Could not create the link right now." };
      revalidatePath("/settings");
      return { token: data.token, expiresAt: data.expires_at };
    }

    export async function revokeShareLink(
      id: string,
    ): Promise<{ ok: true } | { error: string }> {
      const supabase = await createClient();
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) return { error: "Not signed in." };

      const { error } = await supabase
        .from("share_links")
        .update({ revoked_at: new Date().toISOString() })
        .eq("id", id)
        .eq("owner_id", user.id);

      if (error) return { error: "Could not revoke the link right now." };
      revalidatePath("/settings");
      return { ok: true };
    }

    export async function listActiveShareLinks(): Promise<ShareLink[]> {
      const supabase = await createClient();
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) return [];

      const { data } = await supabase
        .from("share_links")
        .select("id, token, owner_id, share_type, expires_at, revoked_at, created_at")
        .eq("owner_id", user.id)
        .is("revoked_at", null)
        .gt("expires_at", new Date().toISOString())
        .order("created_at", { ascending: false });

      return (data ?? []) as ShareLink[];
    }
    ```

    ```tsx
        .select("id, token, owner_id, share_type, expires_at, revoked_at, created_at")
        .eq("token", token)
        .is("revoked_at", null)
        .gt("expires_at", new Date().toISOString())
        .maybeSingle();

      if (error || !link) return <NotActive />;

      const typedLink = link as ShareLink;
      // D-07 — owner ID comes from the row, NEVER from query params
      const ownerId = typedLink.owner_id;

      if (typedLink.share_type === "parent_summary") {
        const summary = await buildParentSummary(supabase, ownerId, typedLink.expires_at);
        return <ParentSummaryView summary={summary} />;
      }

      const snapshot = await buildTeacherSnapshot(supabase, ownerId, typedLink.expires_at);
      return <TeacherSnapshotView snapshot={snapshot} />;
    }

    function NotActive() {
      return (
        <main className="mx-auto max-w-md p-8">
          <div className="rounded-xl border border-border bg-card p-6 text-center">
            <h1 className="text-lg font-semibold">This link is no longer active</h1>
            <p className="mt-2 text-sm text-muted">
              The student who shared this with you may have revoked it, or it may have reached the end of its 7-day window.
              Ask them for a new link if you need one.
            </p>
          </div>
        </main>
      );
    }

    async function buildParentSummary(
      supabase: NonNullable<ReturnType<typeof createServiceClient>>,
      ownerId: string,
      expiresAt: string,
    ): Promise<ParentSummary> {
      // D-08 — week start: Monday 00:00 UTC for simplicity (timezone refinement deferred)
      const now = new Date();
      const day = now.getUTCDay(); // 0=Sun .. 6=Sat
      const daysFromMonday = (day + 6) % 7;
      const weekStart = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), now.getUTCDate() - daysFromMonday));
      const weekStartIso = weekStart.toISOString();
      const nowIso = now.toISOString();
      const next7Iso = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000).toISOString();

      // D-09 — completed this week from task_signals
      const { count: completedThisWeek } = await supabase
        .from("task_signals")
        .select("id", { count: "exact", head: true })
        .eq("owner_id", ownerId)
        .eq("kind", "completed")
        .gte("created_at", weekStartIso);

      // Upcoming next 7 days: assignments with due_at in the window AND not yet done/submitted/graded
      const { count: upcomingNext7Days } = await supabase
        .from("assignments")
        .select("id", { count: "exact", head: true })
        .eq("owner_id", ownerId)
        .gte("due_at", nowIso)
        .lte("due_at", next7Iso)
        .not("status", "in", "(done,submitted,graded)");

      // D-10 — study time from time_logs (closed sessions only)
      const { data: logs } = await supabase
        .from("time_logs")
        .select("opened_at, closed_at")
        .eq("owner_id", ownerId)
        .gte("opened_at", weekStartIso)
        .not("closed_at", "is", null);

      const totalMs = (logs ?? []).reduce((acc, l) => {
        const o = l.opened_at ? new Date(l.opened_at).getTime() : 0;
        const c = l.closed_at ? new Date(l.closed_at).getTime() : 0;
        return c > o ? acc + (c - o) : acc;
      }, 0);

      return {
        completedThisWeek: completedThisWeek ?? 0,
        upcomingNext7Days: upcomingNext7Days ?? 0,
        studyMinutesThisWeek: Math.round(totalMs / 60000),
        weekStartIso,
        expiresAt,
      };
    }

    async function buildTeacherSnapshot(
      supabase: NonNullable<ReturnType<typeof createServiceClient>>,
      ownerId: string,
      expiresAt: string,
    ): Promise<TeacherSnapshot> {
      const { data: classRows } = await supabase
        .from("classes")
        .select("name, ai_mode")
        .eq("owner_id", ownerId)
        .order("name");

      const { data: profile } = await supabase
        .from("profiles")
        .select("reading_font, disability_flags, extra_time_pct")
        .eq("id", ownerId)
        .maybeSingle();

      const dyslexia = Array.isArray(profile?.disability_flags)
        ? (profile?.disability_flags as string[]).includes("dyslexia")
        : false;

      const classes: TeacherClassRow[] = (classRows ?? []).map((c) => ({
        name: c.name ?? "Class",
        aiMode: (c.ai_mode as "red" | "yellow" | "green") ?? "green",
      }));

      return {
        classes,
        readingFont: profile?.reading_font ?? "system",
        extendedReadingTime: dyslexia,
        extraTimePct: profile?.extra_time_pct ?? 0,
        expiresAt,
      };
    }
    ```
