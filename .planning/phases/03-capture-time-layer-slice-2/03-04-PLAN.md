**Step 1 — Add saveIntention to app/(app)/assignments/[id]/actions.ts**

Append (do NOT replace any existing exports) the following server action:

```typescript
const IntentionInput = z.object({
  assignmentId: z.string().uuid(),
  cueValue:     z.string().min(1).max(500),
  cueType:      z.enum(["time", "event", "location", "other"]).default("other"),
  actionText:   z.string().max(500).optional(),
});

export async function saveIntention(
  input: z.infer<typeof IntentionInput>
): Promise<{ ok: true } | { ok: false; error: string }> {
  const parsed = IntentionInput.safeParse(input);
  if (!parsed.success) return { ok: false, error: "Invalid input." };

  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { ok: false, error: "Not signed in." };

  const { error } = await supabase
    .from("assignment_intentions")
    .insert({
      owner_id:      user.id,
      assignment_id: parsed.data.assignmentId,
      cue_type:      parsed.data.cueType,
      cue_value:     parsed.data.cueValue,
      action_text:   parsed.data.actionText ?? null,
    });

  if (error) return { ok: false, error: error.message };
  return { ok: true };
}
```

**Step 2 — Create app/(app)/assignments/[id]/intention-prompt.tsx**

Implement exactly as specified in research §Pattern 6. Key requirements:

1. `useEffect(() => { router.replace(pathname); }, [router, pathname])` — runs on mount immediately (Pitfall 6 guard — URL cleaned before user can refresh)
2. Two buttons: "Save" and "Skip for now" — both result in no visible state change after action (the URL is already cleaned by useEffect)
3. Skip function: no server call, no navigation — URL already cleaned, component just becomes dormant
4. Save function: calls `saveIntention({ assignmentId, cueValue: cue.trim() })` via useTransition
5. After save: show a brief success state "Got it." then fade (use state: "idle" | "saving" | "saved")
6. Textarea placeholder: "After dinner, at my desk" or "Study hall tomorrow" (copy from research)
7. Section label: "When and where will you start?" (not a question mark-heavy interrogation)
8. Sub-label: "For example: 'After dinner, at my desk' or 'Study hall tomorrow.'"
9. No red colors. No required field validation. Skip is always available.

```typescript
"use client";
import { useEffect, useState, useTransition } from "react";
import { useRouter, usePathname } from "next/navigation";
import { saveIntention } from "./actions";

export function IntentionPrompt({ assignmentId }: { assignmentId: string }) {
  const router   = useRouter();
  const pathname = usePathname();
  const [pending, startTransition] = useTransition();
  const [cue, setCue]     = useState("");
  const [status, setStatus] = useState<"idle" | "saved">("idle");

  // Pitfall 6 guard: remove ?intent=new from URL immediately on mount
  useEffect(() => {
    router.replace(pathname);
  }, [router, pathname]);

  function skip() {
    // URL already cleaned by useEffect — nothing else needed
  }

  function save() {
    if (!cue.trim()) return;
    startTransition(async () => {
      const result = await saveIntention({ assignmentId, cueValue: cue.trim() });
      if (result.ok) setStatus("saved");
    });
  }

  if (status === "saved") {
    return (
      <section className="rounded-2xl border border-border bg-card p-5">
        <p className="text-sm text-muted">Got it. You've got a plan.</p>
      </section>
    );
  }

  return (
    <section className="rounded-2xl border border-border bg-card p-5 space-y-3">
      <p className="font-medium">When and where will you start?</p>
      <p className="text-sm text-muted">
        For example: "After dinner, at my desk" or "Study hall tomorrow."
      </p>
      <textarea
        value={cue}
        onChange={(e) => setCue(e.target.value)}
        rows={2}
        className="w-full rounded-lg border border-border bg-card px-3 py-2 text-sm"
        placeholder="After dinner, at my desk..."
      />
      <div className="flex gap-2">
        <button
          type="button"
          onClick={save}
          disabled={pending || !cue.trim()}
          className="rounded-md bg-accent px-3 py-1.5 text-sm font-medium text-white disabled:opacity-50"
        >
          {pending ? "Saving…" : "Save"}
        </button>
        <button
          type="button"
          onClick={skip}
          className="rounded-md border border-border px-3 py-1.5 text-sm text-muted hover:bg-border/30"
        >
          Skip for now
        </button>
      </div>
    </section>
  );
}
```
